    inner sep=0
  },
  terminator/.style={
    feature,
    shape=terminator,
    anchor=south,
    yshift=0.5\pgflinewidth,
    minimum height=2mm,
    minimum width=2mm,
    inner sep=0
  },
  line cap=rect,
}
